# 一、if判断

```javascript
if( 条件 ){
   code // 这是条件 为真的时候执行该代码
} 

//case 1
var a = 3,
b = 5;
if ( a < b ){
	alert(1);//1
}
//case 2 不要做奇怪的比较
var a = "3",
b = "5";
if ( a < b ){
	alert(1);//会弹出1，但此时比较的是3和50的首位的unicode码;如果a=6,就不弹出了，也不报错
}
//case 3
var a = "false";
if(a){
    alert(1);//弹出1,这是字符串，只有空字符串是false
}
```

如果（）括号里面是`true`则运行{} 中代码

```javascript
if( 条件 ){ 
	code 1 // 这是条件 满足时执行的代码
}else{ 
	code 2 // 这是条件 不满足时执行的代码
}
```

如果（）括号里面是`false` 则运行 else 大括号中的 `code 2`—code1/code2总有一个会运行

```javascript
 if( 条件 ){ 
	code 1 // 这是条件一 满足时执行的代码
}else if(  条件二 ){
	code 2 //这是条件二 满足时执行的代码
}else{ code 3 }    

//case 1
var a = 6;
if(a>10){
   	alert("a大于10");
}else if (a>5){
    alert("a大于5")；//只会弹出a大于5，执行到此处就结束了
}else if (a>2){
    alert("a大于4")；
}

var b = 10;
if(b>13){
	alert("b小于11");
}else if(b>11){
	alert("b小于11");
}else {
	alert("前面都不满足");
}
```

如果（）括号里面是`false` 则运行 dlse 大括号中的 `code 2`—code1/code2总有一个会运行

> if()中的条件会**强制转化为布尔值**，为false数据: `false` `0` `''` `null` `undefined` `NaN`

# 二、 三目运算

```javascript
var a = 10;
var b = 20;
if (a<b){
    alert(1);
}else{
    alert(0);
}
//等价于
a<b ? alert(1) : alert(0);//1
```

```
//case
#wrap {
  width: 100px;
  height: 200px;
}
#wrap.red {
  background-color: red;
}
#wrap.pink {
  background-color: pink;
}
...
<div id="wrap" class="red"></div>
...
var oWrap = document.getElementById("wrap");
var bool = true;
oWrap.onclick = function () {
if (bool){
//方法一
     if (bool){
        this.className = "pink";
        bool = false;
      }else{
         this.className = "red";
       	 bool = true;
      }
//方法二        
        bool ? this.className = "pink" : this.className = "red";
        bool = !bool;
//方法三，更好
    this.className = bool ? "pink" : "red";
    bool = !bool;
};
```

# 三、switch 条件语句

```javascript
var name = "阿飞";
switch (name) {
    case "阿飞":
        alert("帅");
        break;//跳出switch
    case "朝哥":
        alert("丑");
        break;
    case "心艾":
        alert("好");
        break;//当条件1满足，既要执行1的内容也要执行2的内容，满足条件2，只执行2，那么可以不写条件1的break
        //或者两种条件执行同一个内容的时候，不写break
    default:
        alert("无法识别");
        break;
}
```

- `break` break 语句。它用于跳出 switch() 语句，来阻止代码自动地向下一个 case 运行
- `default` 关键词来规定匹配不存在时做的事情

```
//点击切换颜色的升级版
#wrap {
            width: 100px;
            height: 200px;
}
#wrap.red {
            background-color: red;
}
#wrap.pink {
            background-color: pink;
}
#wrap.green {
            background-color: green;
}
#wrap.yellow {
            background-color: yellow;
}
#wrap.blue {
            background-color: blue;
}
...
<div id="wrap" class="red"></div>
...
var oWrap = document.getElementById("wrap");
var index = 0;
            
oWrap.onclick = function () {
switch (index){
       case 0:
       	this.className = "pink";
       	break;
       case 1:
         this.className = "green";
         break;
       case 2:
          this.className = "blue";
          break;
        case 3:
          this.className = "yellow";
          break;
        case 4:
          this.className = "red";
          break;
         }
          index ++;
          index %= 5;//index=index%5 这种方法更好,效率比if高
                // if (index>4) {
                //     index = 0;
                // }
};

//更好的解决方法
var oWrap = document.getElementById("wrap");
var index = 0;
var nameArr = ["pink","green","blue","yellow","red"];

oWrap.onclick = function () {
	this.className = nameArr[index];
};
    index ++;
    index %= 5;
//或者
oWrap.onclick = function () {
	this.className = nameArr[index];
	index = (index+1)%5;
};
//再或者
oWrap.onclick = function () {
	this.className = nameArr[index ++];//先赋值，再累加
	index %= 5; 
};
```

```javascript
//逻辑运算代替if
var a = 10;

if ( a===10 ) {
     a ++;
}

//等价于
(a===10) && a++;//流程控制，遇到假就停

alert(a);
```
